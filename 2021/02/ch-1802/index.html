CH 1802 表达式计算4

  2021-02-06   题解   栈  评论

链接

摘要

   求中缀表达式的计算值

解

   先转为后缀表达式后计算    转换过程如下:    两个栈, 一个作暂存容器, 一个作结果存储    数字直接入结果栈, 小技巧:在数字结束后加 $#$, 方便之后将字符转化为数字    符号入暂存栈, 入前需要考虑优先级, 优先级在事先设置为函数或数组, 若栈中优先级更大, 则先弹出后入栈, 这样就能保证暂存栈中优先级单调递增, 之后弹出至结果栈时顺序颠倒, 即成为后缀表达式的运算顺序    左括号直接入栈    若为右括号, 将左括号至上的所有符号连同左括号弹出    
   后缀表达式的计算:    取数并得出数字, 遇到符号取两次堆顶计算再入堆, 最终堆顶即为结果

debug

   1.判数字是 && 而非 ||    2.判优先级时注意暂存栈是否已空

代码

#include<cstdio>
#include<cstdlib>
#include<cstring>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

const int N=6e1+5;
char a[N],s1[N],s2[N];
int s3[N],c1,c2,c3;

inline int level(char x)
{
    if(x=='*'||x=='/')
        return 2;
    else if(x=='+'||x=='-')
        return 1;
    else if(x=='^')
        return 3;
    return 0;
}

inline int quickpow(int x,int p)
{
    int res=1;
    while(p)
    {
        if(p&1)
            res*=x;
        x*=x,p>>=1;
    }
    return res;
}

int main()
{
    scanf("%s",a+1);
    int l=strlen(a+1);
    ifor(i,1,l)
    {
        if(a[i]<='9'&&a[i]>='0')
            {s2[++c2]=a[i];continue;}
        if(a[i-1]<='9'&&a[i-1]>='0')
            s2[++c2]='#';
        if(a[i]=='(')
            s1[++c1]=a[i];
        else if(a[i]==')')
        {
            while(s1[c1]!='(')
                s2[++c2]=s1[c1--];
            c1--;
        }
        else 
        {
            while(level(a[i])<=level(s1[c1])&&c1)
                s2[++c2]=s1[c1--];
            s1[++c1]=a[i];
        }
    }
    if(a[l]<='9'&&a[l]>='0')
        s2[++c2]='#';
    while(c1)
        s2[++c2]=s1[c1--];
    int tmp=0;
    ifor(i,1,c2)
    {
        if(s2[i]>='0'&&s2[i]<='9')
        {
            tmp=(tmp<<1)+(tmp<<3)+(s2[i]^48);
            continue;
        }
        if(s2[i]=='#')
            s3[++c3]=tmp,tmp=0;
        else
        {
            c3--;
            if(s2[i]=='+')
                s3[c3]+=s3[c3+1];
            else if(s2[i]=='-')
                s3[c3]-=s3[c3+1];
            else if(s2[i]=='*')
                s3[c3]*=s3[c3+1];
            else if(s2[i]=='/')
                s3[c3]/=s3[c3+1];
            else if(s2[i]=='^')
                s3[c3]=quickpow(s3[c3],s3[c3+1]);
            else ++c3;
        }
    }
    printf("%d",s3[c3]);
    system("pause");
    return 0;
}

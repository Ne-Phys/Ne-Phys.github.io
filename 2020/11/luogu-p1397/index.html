Luogu P1397 [NOI2013]矩阵游戏

  2020-11-14   题解   矩阵乘法  评论

链接

摘要

   根据递推式求 $F[n][m]$ $mod$ $1e9$$+$$7$
   数据范围见原题

解

   由递推式, 很容易想到构造使用矩阵乘法

技巧

1.扩展欧拉定理

   $a^b$ $≡$ $a^{b\ mod\ φ(n)}$ $(mod$ $n)$ $(a$与$n$互质$)$
   对指数、底数取模, 优化复杂度

2.十进制快速幂

   本题中, 为应对高精度指数(未使用扩展欧拉定理)

inline node qkpw(node bs,char *p)
{
    node res;
    res.a[1][1]=res.a[2][2]=1;
    ll l=strlen(p+1);
    for(ll i=l;i>=1;i--)
    {
        node t=bs;
        if(p[i]=='9')
        {   t=t*t,t=t*t,t=t*t,t=t*bs;
            res=res*t;}
        else if(p[i]=='8')
        {   t=t*t,t=t*t,t=t*t;
            res=res*t;}
        else
        {   ifor(j,1,p[i]-'0')
            {   res=res*bs;}}
        node k=bs;
        bs=bs*bs,bs=bs*bs,bs=bs*bs,bs=bs*k,bs=bs*k;
    }
    return res;
}

#include<cstdio>
#include<cstdlib>
#include<cstring>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

typedef long long ll;
const ll inf=1e6+5,mod=1e9+7;
ll n,m,a,b,c,d,e;
char ns[inf],ms[inf];

inline ll md(ll x)
{   return x>=mod?x-mod:x;}

struct node
{
    ll a[3][3];
    node()
    {   memset(a,0,sizeof(a));}
    inline node operator*(const node&x)
    const
    {
        node res;
        res.a[1][1]=md(a[1][1]*x.a[1][1]%mod+a[1][2]*x.a[2][1]%mod);
	    res.a[1][2]=md(a[1][1]*x.a[1][2]%mod+a[1][2]*x.a[2][2]%mod);
	    res.a[2][1]=md(a[2][1]*x.a[1][1]%mod+a[2][2]*x.a[2][1]%mod);
	    res.a[2][2]=md(a[2][1]*x.a[1][2]%mod+a[2][2]*x.a[2][2]%mod);
        return res;
    }
}ans,bs1,bs2,bs3;

inline node qkpw(node bs,ll p)
{
    node res;
    res.a[1][1]=res.a[2][2]=1;
    while(p)
    {
        if(p&1)
        {   res=res*bs;}
        bs=bs*bs,p>>=1;
    }
    return res;
}

POJ 3974 Palindrome

  2020-12-22   题解   字符串Hash manacher  评论

链接

摘要

   字符串 $s$ 的最长回文子串

解

   $O(nlogn)$
   关于字符串$Hash$:
   视作 $P$ 进制数, $P$ 取 $131$, $13331$ 冲突最少
   ull 自然取模, 规避模运算的高复杂度
   求子串的 $Hash$ 类似前缀和的思想 $s[j]-s[i-1]*$$P^{j-i+1}$
   分奇偶两种情况, 枚举中间点, 二分回文半径
   
   $O(n)$
   $manacher$ 算法
   对奇偶性的处理: 在字符间添加 '#'
   对边界的处理: $s[0]=$ '!', $s[l+2]=$ ‘^’ $(s[l]=a[n],s[1]=s[l+1]=$ '#' $)$
   维护当前最大回文串的中心字符下标 $mid$ 、回文串右边界下标
   遍历字符串, 若下标 $i$$<$$r$, 则可运用中点公式查找关于 $mid$ 对称位置的回文半径
   对称位置的回文半径有三种情况:
   1.j-f[j] 未抵左边界, 则直接可以确定 f[i]=f[j](若扩展, 对称后与之前确定的矛盾)
   2.j-f[j] 恰抵左边界, 此时 f[i] 在 f[j] 的基础上扩展
   3.j-f[j] 越过左边界, 则也可以直接确定 f[i]=r-i+1(若扩展, 则非最大回文串)
   在这一基础上进一步分析复杂度:
   for 循环 $n$ 次
   情况1、3: $O(1)$
   情况2:
   while 受 $r$ 制约, $r$ 依据 $i+k$, $i+k$ 在 $r$ 的基础之上递增
   相当于 $r$ 为 $i+k$ 存点, 循环下一个字符时要么读取存点, 要么新增并再存点
   => while 循环 $n$ 次
   复杂度即是 $2n$ $=$ $O(n)$

代码

   字符串 $Hash$

#include<cstdio>
#include<cstring>
#include<cstdlib>
#define max(a,b) ((a)>(b)?(a):(b))
#define min(a,b) ((a)<(b)?(a):(b))
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

typedef unsigned long long ull;
const int N=1e6+5,P=131;
int lnth;
char s[N];
ull a[N],b[N],p[N];

inline bool judge(int x,int l)
{
    ull a1=a[x+l]-a[x]*p[l];
    ull b1=b[lnth-x+1+l]-b[lnth-x+1]*p[l];
    if(a1==b1)
        return true;
    else return false;
}

inline bool judge2(double x,int l)
{
    ull a1=a[(int)(x+l-0.5)]-a[(int)(x-0.5)]*p[l];
    ull b1=b[(int)(lnth-x+0.5+l)]-b[(int)(lnth-x+0.5)]*p[l];
    if(a1==b1)
        return true;
    else return false;
}

int main()
{
    p[0]=1;
    ifor(i,1,1000000)
        p[i]=p[i-1]*P;
    int c=0;
    while(scanf("%s",s+1))
    {
        ++c;
        if(s[1]=='E')
            break;
        lnth=strlen(s+1);
        ifor(i,1,lnth)
        {
            a[i]=a[i-1]*P+(s[i]-'a'+1);
            b[i]=b[i-1]*P+(s[lnth-i+1]-'a'+1);
        }
        int mxn=1;
        ifor(i,1,lnth)
        {
            int r=min(i-1,lnth-i),l=0;
            while(l<r)
            {
                int mid=(l+r+1)>>1;
                if(judge(i,mid))
                    l=mid;
                else r=mid-1;
            }
            mxn=max(mxn,l*2+1);
        }
        for(double i=0.5;i<=lnth-0.5;++i)
        {
            int r=min(i-0.5,lnth-i+0.5),l=0;
            while(l<r)
            {
                int mid=(l+r+1)>>1;
                if(judge2(i,mid))
                    l=mid;
                else r=mid-1;
            }
            mxn=max(mxn,l*2);
        }
        printf("Case %d: %d\n",c,mxn);
    }
    //system("pause");
    return 0;
}

   $manacher$

#include<cstdio>
#include<cstdlib>
#include<cstring>
#define max(a,b) ((a)>(b)?(a):(b))
#define min(a,b) ((a)<(b)?(a):(b))
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

typedef unsigned long long ull;
const int N=2e6+5,P=131;
char s[N],con[N];
int f[N];

int main()
{
    int id=0;
    while(scanf("%s",s+1)&&s[1]!='E')
    {
        id++;
        int l=strlen(s+1);
        con[0]='$';
        int c=0;
        ifor(i,1,l)
            con[++c]='#',con[++c]=s[i];
        con[++c]='#',con[c+1]='^';
        int r=0,mid=0;
        int mxn=1;
        ifor(i,1,c-1)
        {
            if(i<r)
                f[i]=min(f[2*mid-i],r-i+1);
            else f[i]=1;
            while(con[i+f[i]]==con[i-f[i]])
                f[i]++;
            if(r<i+f[i]-1)
                mid=i,r=i+f[i]-1;
            mxn=max(mxn,f[i]-1);
        }
        printf("Case %d: %d\n",id,mxn);
    } 
    system("pause");
    return 0;
}

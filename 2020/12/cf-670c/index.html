Codeforces 670C Cinema

  2020-12-08   题解   离散化  评论

链接

摘要

   $n$ 个人各自会一门语言, 语言用 $int$ 范围内的整数表示
   $m$ 部电影, 各自有配音和字幕
   求一部电影, 其配音的语言其次字幕的语言会的人最多

解

   语言表示在 $int$ 范围内, 故需要离散化
   一是用 $map$ 直接用作数组
   二是手写离散化, 时空复杂度小

代码

   $map$

#include<cstdio>
#include<map>
#include<cstdlib>
#include<algorithm>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

const int N=2e5+5;
int n,m;
std::map<int,int> map;

struct node
{
    int p,a,b;
    inline bool operator<(const node&x)
    const
        {return (a>x.a)||(a==x.a&&b>x.b);}
}mv[N];

int main()
{
    scanf("%d",&n);
    int x;
    ifor(i,1,n)
        scanf("%d",&x),map[x]++;
    scanf("%d",&m);
    ifor(i,1,m)
    {
        scanf("%d",&x);
        mv[i].p=i;
        if(map.count(x))
            mv[i].a=map[x];
    }
    ifor(i,1,m)
    {
        scanf("%d",&x);
        if(map.count(x))
            mv[i].b=map[x];
    }
    std::sort(mv+1,mv+m+1);
    printf("%d",mv[1].p);
    //system("pause");
    return 0;
}

   手写离散化

#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

const int N=2e5+5;
int n,m;
int l[N],t[N],ttl[N],c;

struct node
{
    int p,a,b;
    inline bool operator<(const node&x)
    const
        {return (a>x.a)||((a==x.a)&&(b>x.b));}
}mv[N];

inline void discrete()
{
    std::sort(l+1,l+n+1);
    ifor(i,1,n)
    {
        if(i==1||l[i]!=l[i-1])
            t[++c]=l[i];//t数组用于查位置
        ttl[c]++;
    }
}

inline int query(int x)
{
    return std::lower_bound(t+1,t+c+1,x)-t;
}

int main()
{
    scanf("%d",&n);
    ifor(i,1,n)
        scanf("%d",&l[i]);
    discrete();
    scanf("%d",&m);
    int x;
    ifor(i,1,m)
    {
        scanf("%d",&x);
        mv[i].p=i;
        if(t[query(x)]==x)
            mv[i].a=ttl[query(x)];
    }
    ifor(i,1,m)
    {
        scanf("%d",&x);
        if(t[query(x)]==x)
            mv[i].b=ttl[query(x)];
    }
    std::sort(mv+1,mv+m+1);
    printf("%d",mv[1].p);
    //system("pause");
    return 0;
}

Luogu P1050 循环

  2020-03-23   题解   模拟 递推  评论

链接

摘要

   对于一个整数 $n$ 的正整数次幂，后 $k$ 位是否发生循环
   若是 求循环长度
   $k$ $\leq$ $100$

解

   从尾部递推
   依次保证 后 $1$ 位循环，后 $2$ 位循环……后 $k$ 位循环
   定义fac在保证原数与之相乘 后 $m$ 位不变的前提下，探究保证第 $m+1$ 位不变的条件
   定义fac_maker以计算fac
   若次数 $cnt > 10$ 可判断不循环
   若 $cnt < 10$，将每一位探究的次数相乘可得到循环长度
   值得注意的是 由数据范围知应使用高精度

debug

  1. 乘法高精度中c[i+j-1]=a[i]*b[j] c[i+j-1]+=a[i]*b[j]

代码

#include<cstdio>
#include<cstdlib>
#include<cstring>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

const int inf=1e4+9,times[10]={1,1,4,4,2,1,1,4,4,2};
int k;
int ans[102],ori[102],maiN[inf],fac[inf],fac_maker[inf],tmp[inf];
char s[102];

inline void mul_1(int a[],int b[],int c[])
{
    ifor(i,1,k)
        ifor(j,1,k)
            c[i+j-1]+=a[i]*b[j],c[i+j]+=c[i+j-1]/10,c[i+j-1]%=10;
}

inline void mul_2(int a[],int b,int c[])
{
    ifor(i,1,k)
        c[i]+=a[i]*b,c[i+1]+=c[i]/10,c[i]%=10;
}

int main()
{
    scanf("%s%d",s+1,&k);
    int l=strlen(s+1);
    for(int i=l;i>=l-k+1;--i)
        ori[l-i+1]=s[i]-'0';
    ifor(i,1,k) maiN[i]=ori[i];
    ifor(i,1,times[ori[1]]-1)
    {
        memset(tmp,0,sizeof(tmp));
        mul_1(maiN,ori,tmp);
        ifor(j,1,k)
            maiN[j]=tmp[j];
    }
    ans[1]=times[ori[1]];
    ifor(i,1,k)
        fac[i]=fac_maker[i]=maiN[i];
    int cur=1;
    while(++cur<k+1)
    {
        ifor(i,1,k)
            maiN[i]=ori[i];
        int cnt=0;
        while(cnt<=10)
        {
            memset(tmp,0,sizeof(tmp));
            mul_1(maiN,fac,tmp);
            ++cnt;
            if(tmp[cur]==ori[cur]) break;
            ifor(i,1,k)
                maiN[i]=tmp[i];
            memset(tmp,0,sizeof(tmp));
            mul_1(fac_maker,fac,tmp);
            ifor(i,1,k)
                fac_maker[i]=tmp[i];
        }
        if(cnt>10) return puts("-1"),0;
        memset(tmp,0,sizeof(tmp));
        mul_2(ans,cnt,tmp);
        ifor(i,1,100) 
            ans[i]=tmp[i];
        ifor(i,1,k) fac[i]=fac_maker[i];
    }
    int kk=100;
    while(ans[kk]==0&&kk>1) --kk;
    for(int i=kk;i>=1;--i)
        printf("%d",ans[i]);
    system("pause");
    return 0;
}

Luogu P2114 [NOI2014]起床困难综合症

  2020-03-11   题解   位运算 贪心  评论

链接

摘要

   求数值 $k$$\in$$[1,m]$ 经过 $n$ 次 $op$ $t$ 操作后的最大值

   $2\leq n\leq10^5$    $2\leq m\leq10^9$    $0\leq t\leq10^9$    $op$ 一定为 $or$ $xor$ $and$ 中的一种

解

   定义k1=0x7fffffff k2=0
   从二进制来看 $k1$ 每一位都是 $1$,$k2$ 每一位都是 $0$
   输入的过程中直接对 $k1$ $k2$ 进行操作
   对于第 $i$ 位 遵循贪心原则 $:$

  • 若k2==1则ans+=1<<i 初始值的第 $i$ 位为 $0$
  • 或若k1==1则需考虑若初始值的第 $i$ 位为 $1$ 是否超出初始值范围 若否再而累加

debug

   ans累加ans|=1<<i ans+=1<<i

代码

#include<cstdio>
#include<cstdlib>
#define ifor(i,a,b) for(int i=(a);i<=(b);++i)

int n,m,t,ans,k1=0x7fffffff,k2=0;
char str[3];

int main()
{
    scanf("%d%d",&n,&m);
    ifor(i,1,n)
    {
        scanf("%s%d",str,&t);
        if(str[0]=='O') k1|=t,k2|=t;
        if(str[0]=='X') k1^=t,k2^=t;
        if(str[0]=='A') k1&=t,k2&=t;
    }
    for(int i=29;~i;--i)
    {
        if(k2>>i&1) ans+=1<<i;
        else if(k1>>i&1&&m>=1<<i)
            ans+=1<<i,m-=1<<i;
    }
    printf("%d",ans);
    system("pause");
    return 0;
}
